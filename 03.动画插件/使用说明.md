1. 引入插件（ animate.js ）
2. 配置参数
    ```js
    var animate = new myPlugins.Animate({
            dom: document.querySelector('.demo'),       // 设置动画的目标元素
            style: {     // 动画属性和结束值
                top: 300,
                left: 500,
                opacity: 0.1,
                backgroundColor: 'rgb(0, 0, 255)'
            },
            step: 16,   // 每帧动画时间，单位“毫秒”，默认为16毫秒
            duration: 5000,     // 动画总持续时间，单位“毫秒”，默认为1000毫秒
            moving: function(status) {},     // 获取动画持续中当前位置参数，参数格式为函数，函数中设置形参用于接收返回值
            end: function() {}      // 动画结束时处理函数
        });

        animate.start(function() {      // 动画开始调用start方法，可传入回调函数
            console.log('开始');
        });
        animate.stop(function() {       // 动画暂停调用stop方法，可传入回调函数
            console.log('暂停');
        });
    ```
