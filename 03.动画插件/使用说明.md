1. 引入插件（ animate.js ）
2. 配置参数
    ```js        
    var animate = new myPlugins.Animate({
        begin: {    // 元素动画开始样式
            top: 50,
            left: 50,
            opacity: 1,
            bgColor: { r: 255, g: 0, b: 0 }     // 颜色值属性必须以对象格式分别配置rgb参数值
        },
        end: {     // 元素动画结束样式
            top: 300,
            left: 500,
            opacity: 0.1,
            bgColor: { r: 0, g: 0, b: 255 }     // 颜色值属性必须以对象格式分别配置rgb参数值
        },
        step: 16,   // 每帧动画时间，单位“毫秒”，默认为16毫秒
        duration: 5000,     // 每帧动画时间，单位“毫秒”，默认为16毫秒
        running: function(data) {      // 每帧动画时间，单位“毫秒”，默认为16毫秒
            dom.style.top = data.top + 'px';
            dom.style.left = data.left + 'px';
            dom.style.opacity = data.opacity;
            dom.style.backgroundColor = `rgb(${data.bgColor.r}, ${data.bgColor.g}, ${data.bgColor.b})`;
        },   
        over: function() {       // 动画结束时处理函数，可不进行配置
            console.log('结束');
        },
    });

    animate.start(function() {      // 动画开始调用start方法，可传入回调函数
        console.log('开始');
    });
    animate.stop(function() {       // 动画暂停调用stop方法，可传入回调函数
        console.log('暂停');
    });
    ```
